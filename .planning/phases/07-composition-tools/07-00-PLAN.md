```typescript
// src/store/animationStore.ts (to be created in Wave 1)
interface SplinePoint {
  beat: number    // absolute beat position within loop
  value: number   // 0-100 for size/saturation/lightness; 0-360 for hue
}
interface SplineCurve {
  duration: number   // loop window in beats (free-float)
  points: SplinePoint[]
}
type AnimatableProperty = 'size' | 'hue' | 'saturation' | 'lightness'
type ShapeCurves = Partial<Record<AnimatableProperty, SplineCurve>>
interface AnimationState {
  curves: Record<string, ShapeCurves>  // key = shape.id
  setCurve: (shapeId: string, property: AnimatableProperty, curve: SplineCurve) => void
  removeCurve: (shapeId: string, property: AnimatableProperty) => void
  clearShape: (shapeId: string) => void
}
export const animationStore: StoreApi<AnimationState>
export const useAnimationStore: <T>(selector: (s: AnimationState) => T) => T
```
